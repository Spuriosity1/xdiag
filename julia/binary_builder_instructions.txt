dependencies:

Julia_jll
libjulia_jll
libcxxwrap_julia_jll
MKL_jll
HDF5_jll


commands (local):

JLCXX_DIR=`julia -e "using CxxWrap; ; println(CxxWrap.prefix_path())"`/lib/cmake/JlCxx
cmake -DHYDRA_JULIA_WRAPPER=On -D JlCxx_DIR=$JLCXX_DIR -S . -B build
cmake --build build --verbose -j$nproc
cmake --install build


commands (binary builder):

cd hydra
cmake -DCMAKE_INSTALL_PREFIX=$prefix -DCMAKE_TOOLCHAIN_FILE=${CMAKE_TARGET_TOOLCHAIN} -DCMAKE_BUILD_TYPE=Release -DHYDRA_JULIA_WRAPPER=On -DJlCxx_DIR=$prefix/lib/cmake -S . -B build
cmake --build build --verbose -j$nproc
cmake --install build
